-- Services
local Players          = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService       = game:GetService("RunService")

local Player = Players.LocalPlayer
local Camera = workspace.CurrentCamera

-- Aimlock Variables
local FeatureEnabled  = false
local IsHoldingKey    = false
local Keybind         = Enum.UserInputType.MouseButton2  

local LockedPart      = nil
local BaseSmoothing   = 25           
local CurrentSmoothing = BaseSmoothing

local AimPart         = "HumanoidRootPart"
local UseRandomPart   = true         
local LastRandomSwitch = 0
local RANDOM_SWITCH_INTERVAL = 0.3

local WallCheck       = false   -- ← Re-added

local PossibleAimParts = {"Head", "HumanoidRootPart", "UpperTorso", "LowerTorso"}

local RayParams = RaycastParams.new()
RayParams.FilterType = Enum.RaycastFilterType.Exclude

local function updateRaycastFilter()
    if Player.Character then
        RayParams.FilterDescendantsInstances = {Player.Character}
    end
end

Player.CharacterAdded:Connect(updateRaycastFilter)
updateRaycastFilter()  

local lastVariationUpdate = 0
local VARIATION_INTERVAL = 0.25
local EXTRA_MIN = 5
local EXTRA_MAX = 10

local function UpdateDynamicSmoothing()
    local now = tick()
    if now - lastVariationUpdate >= VARIATION_INTERVAL then
        local extra = math.random() * (EXTRA_MAX - EXTRA_MIN) + EXTRA_MIN
        CurrentSmoothing = BaseSmoothing + extra
        lastVariationUpdate = now
    end
end

local function GetCurrentAimPart()
    if not UseRandomPart then return AimPart end
    local now = tick()
    if now - LastRandomSwitch >= RANDOM_SWITCH_INTERVAL then
        AimPart = PossibleAimParts[math.random(1, #PossibleAimParts)]
        LastRandomSwitch = now
    end
    return AimPart
end

local function HasLineOfSight(targetPart)
    if not WallCheck then return true end
    if not targetPart then return false end

    local origin = Camera.CFrame.Position
    local direction = (targetPart.Position - origin)
    local rayResult = workspace:Raycast(origin, direction, RayParams)

    if rayResult and rayResult.Instance then
        local hitModel = rayResult.Instance:FindFirstAncestorWhichIsA("Model")
        if hitModel then
            local hitPlayer = Players:GetPlayerFromCharacter(hitModel)
            if hitPlayer and hitPlayer == Players:GetPlayerFromCharacter(targetPart.Parent) then
                return true
            end
        end
        return false
    end
    return true
end

local function LockTarget()
    LockedPart = nil

    local mousePos = UserInputService:GetMouseLocation()
    local unitRay = Camera:ScreenPointToRay(mousePos.X, mousePos.Y)

    local rayResult = workspace:Raycast(unitRay.Origin, unitRay.Direction * 1000, RayParams)

    if rayResult and rayResult.Instance then
        local model = rayResult.Instance:FindFirstAncestorWhichIsA("Model")
        if model then
            local hum = model:FindFirstChildOfClass("Humanoid")
            local targetPlr = Players:GetPlayerFromCharacter(model)
            if hum and hum.Health > 0 and targetPlr and targetPlr ~= Player then
                if not (targetPlr.Team == Player.Team and targetPlr.Team ~= nil) then
                    local aimTarget = model:FindFirstChild(GetCurrentAimPart()) or model:FindFirstChild("HumanoidRootPart")
                    if aimTarget and aimTarget:IsA("BasePart") and HasLineOfSight(aimTarget) then
                        LockedPart = aimTarget
                        return true
                    end
                end
            end
        end
    end

    -- fallback to closest visible target (with wall check)
    local closestPart = nil
    local shortest = math.huge
    local mouseScreen = UserInputService:GetMouseLocation()

    for _, plr in Players:GetPlayers() do
        if plr ~= Player and plr.Character then
            local humanoid = plr.Character:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 and not (plr.Team == Player.Team and plr.Team ~= nil) then
                local part = plr.Character:FindFirstChild(GetCurrentAimPart()) or plr.Character:FindFirstChild("HumanoidRootPart")
                if part and part:IsA("BasePart") and HasLineOfSight(part) then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(part.Position)
                    if onScreen then
                        local dist = (Vector2.new(screenPos.X, screenPos.Y) - mouseScreen).Magnitude
                        if dist < shortest then
                            shortest = dist
                            closestPart = part
                        end
                    end
                end
            end
        end
    end

    if closestPart then
        LockedPart = closestPart
        return true
    end
    return false
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.UserInputType == Keybind or input.KeyCode == Keybind then
        IsHoldingKey = true
        if FeatureEnabled and not LockedPart then LockTarget() end
    end
end)

UserInputService.InputEnded:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.UserInputType == Keybind or input.KeyCode == Keybind then
        IsHoldingKey = false
        LockedPart = nil
    end
end)

local mousemoverelFunc = getfenv().mousemoverel or getgenv().mousemoverel

-- ==================== SILHOUETTE ESP (Highlight + Name Tags) ====================

local ESPEnabled = false
local ShowNameTags = true
local ESPOutlineColor = Color3.fromRGB(255, 165, 0)
local ESPFillTransparency = 0.85

local ESPHighlights = {}
local NameTags = {}

local function CreateESPForPlayer(plr)
    if not plr.Character then return end

    local highlight = Instance.new("Highlight")
    highlight.Name = "SilhouetteESP"
    highlight.Adornee = plr.Character
    highlight.FillTransparency = ESPFillTransparency
    highlight.OutlineColor = ESPOutlineColor
    highlight.OutlineTransparency = 0
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Enabled = true
    highlight.Parent = plr.Character

    ESPHighlights[plr.UserId] = highlight

    local billboard = Instance.new("BillboardGui")
    billboard.Name = "NameESP"
    billboard.Adornee = plr.Character:FindFirstChild("Head") or plr.Character:FindFirstChild("HumanoidRootPart")
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.AlwaysOnTop = true
    billboard.LightInfluence = 0
    billboard.Enabled = ShowNameTags
    billboard.Parent = plr.Character

    local text = Instance.new("TextLabel")
    text.Size = UDim2.new(1, 0, 1, 0)
    text.BackgroundTransparency = 1
    text.Text = plr.Name
    text.TextColor3 = Color3.fromRGB(255, 255, 255)
    text.TextStrokeTransparency = 0
    text.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    text.Font = Enum.Font.GothamBold
    text.TextSize = 16
    text.Parent = billboard

    NameTags[plr.UserId] = billboard
end

local function UpdateESP()
    if not ESPEnabled then return end

    for _, plr in ipairs(Players:GetPlayers()) do
        if plr == Player then continue end

        local char = plr.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")

        if not char or not hum or hum.Health <= 0 then
            if ESPHighlights[plr.UserId] then
                ESPHighlights[plr.UserId]:Destroy()
                ESPHighlights[plr.UserId] = nil
            end
            if NameTags[plr.UserId] then
                NameTags[plr.UserId]:Destroy()
                NameTags[plr.UserId] = nil
            end
            continue
        end

        if plr.Team == Player.Team and plr.Team ~= nil then
            if ESPHighlights[plr.UserId] then ESPHighlights[plr.UserId].Enabled = false end
            if NameTags[plr.UserId] then NameTags[plr.UserId].Enabled = false end
            continue
        end

        if not ESPHighlights[plr.UserId] then
            CreateESPForPlayer(plr)
        else
            local hl = ESPHighlights[plr.UserId]
            if hl.Adornee ~= char then hl.Adornee = char end
            hl.FillTransparency = ESPFillTransparency
            hl.OutlineColor = ESPOutlineColor
            hl.Enabled = true

            if NameTags[plr.UserId] then
                NameTags[plr.UserId].Enabled = ShowNameTags
            end
        end
    end
end

local function ClearAllESP()
    for _, hl in pairs(ESPHighlights) do if hl then hl:Destroy() end end
    for _, tag in pairs(NameTags) do if tag then tag:Destroy() end end
    ESPHighlights = {}
    NameTags = {}
end

local ESPConnection
local function StartESPLoop()
    if ESPConnection then return end
    ESPConnection = RunService.RenderStepped:Connect(UpdateESP)
end

local function StopESPLoop()
    if ESPConnection then
        ESPConnection:Disconnect()
        ESPConnection = nil
    end
    ClearAllESP()
end

-- ==================== AIMLOCK LOOP ====================

local AimlockLoop = RunService.RenderStepped:Connect(function()
    if not FeatureEnabled then return end
    if not IsHoldingKey then return end
    if not LockedPart or not LockedPart.Parent then return end

    UpdateDynamicSmoothing()

    local currentAimPartName = GetCurrentAimPart()
    local targetPart = LockedPart.Parent:FindFirstChild(currentAimPartName) or LockedPart.Parent:FindFirstChild("HumanoidRootPart")
    if not targetPart or not targetPart:IsA("BasePart") then return end

    local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
    if not onScreen or screenPos.Z < 0 then return end

    local mouseLoc = UserInputService:GetMouseLocation()
    local moveDivisor = CurrentSmoothing / 5
    local deltaX = (screenPos.X - mouseLoc.X) / moveDivisor
    local deltaY = (screenPos.Y - mouseLoc.Y) / moveDivisor

    if mousemoverelFunc then
        mousemoverelFunc(deltaX, deltaY)
    end

    UpdateESP()
end)
